\sectiontitle{FITUR UTAMA OMNIFLOW PURCHASING}

\noindent
\begin{minipage}[t]{0.485\textwidth}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{OrangePrimary} 1. Purchase Order (PO)}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Nomor PO unik otomatis (PO-YYYY-XXXXX)\par\vspace{2pt}%
    \textbullet\ Terhubung ke supplier di modul XRM\par\vspace{2pt}%
    \textbullet\ Status lengkap: draft, pending\_approval, approved, ordered, fulfilled, cancelled\par\vspace{2pt}%
    \textbullet\ Tanggal pengiriman yang diharapkan (expected delivery)\par\vspace{2pt}%
    \textbullet\ Soft delete --- PO tidak hilang, hanya diarsip}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{OrangePrimary} 2. Approval Workflow}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Assign approver per PO sebelum diproses\par\vspace{2pt}%
    \textbullet\ Timestamp approved\_at tercatat otomatis\par\vspace{2pt}%
    \textbullet\ Catatan approval/rejection (approval\_note)\par\vspace{2pt}%
    \textbullet\ Status rejected dengan alasan tercatat\par\vspace{2pt}%
    \textbullet\ Tidak ada PO diproses tanpa approval}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{OrangePrimary} 3. Fulfillment Tracking}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Tandai PO telah dipesan ke supplier (ordered\_at)\par\vspace{2pt}%
    \textbullet\ Tandai barang telah diterima (fulfilled\_at)\par\vspace{2pt}%
    \textbullet\ Catat siapa yang konfirmasi penerimaan\par\vspace{2pt}%
    \textbullet\ Catatan fulfillment untuk quality check}%

\end{minipage}%
\hfill%
\begin{minipage}[t]{0.485\textwidth}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{OrangePrimary} 4. Invoice Pembelian}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Invoice supplier tertaut ke PO (1:1)\par\vspace{2pt}%
    \textbullet\ Nomor invoice, tanggal, jumlah, jatuh tempo\par\vspace{2pt}%
    \textbullet\ Status pembayaran: unpaid, paid, cancelled\par\vspace{2pt}%
    \textbullet\ Referensi pembayaran (nomor transfer)\par\vspace{2pt}%
    \textbullet\ Upload file invoice (tersimpan di S3)\par\vspace{2pt}%
    \textbullet\ Timestamp paid\_at tercatat otomatis}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{OrangePrimary} 5. Manajemen Lampiran}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Upload dokumen ke PO atau invoice\par\vspace{2pt}%
    \textbullet\ Kategori: quotation, proposal, agreement, invoice, receipt, photo\par\vspace{2pt}%
    \textbullet\ Metadata: nama file, tipe MIME, ukuran, S3 key\par\vspace{2pt}%
    \textbullet\ Soft delete --- arsip, tidak hilang permanen\par\vspace{2pt}%
    \textbullet\ Pencarian lampiran per entitas (PO/Invoice)}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{OrangePrimary} 6. Integrasi Supplier (XRM)}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Pilih supplier langsung dari database XRM\par\vspace{2pt}%
    \textbullet\ Data supplier (kontak, alamat, rekening) otomatis terisi\par\vspace{2pt}%
    \textbullet\ Histori pembelian tersimpan di profil supplier\par\vspace{2pt}%
    \textbullet\ Analitik pengeluaran per supplier}%

\end{minipage}

\vspace{8pt}
