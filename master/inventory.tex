\sectiontitle{FITUR UTAMA OMNIFLOW INVENTORY}

\noindent
\begin{minipage}[t]{0.485\textwidth}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{CyanPrimary} 1. Master Produk \& Varian}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ SKU \& barcode unik per produk\par\vspace{2pt}%
    \textbullet\ Kategori bersarang (parent-child) dengan slug SEO\par\vspace{2pt}%
    \textbullet\ Varian produk (ukuran, warna, dll) per SKU induk\par\vspace{2pt}%
    \textbullet\ Multi-unit of measure (UoM) dengan konversi satuan\par\vspace{2pt}%
    \textbullet\ Harga beli, harga jual, diskon maksimum\par\vspace{2pt}%
    \textbullet\ Min/max stok, status taxable per produk}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{CyanPrimary} 2. Multi-Outlet \& Gudang}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Daftarkan outlet dan gudang terpisah\par\vspace{2pt}%
    \textbullet\ Assign staf ke outlet tertentu (outlet\_users)\par\vspace{2pt}%
    \textbullet\ Saldo stok tercatat per kombinasi produk-lokasi\par\vspace{2pt}%
    \textbullet\ Mutasi stok teraudit: masuk, keluar, koreksi\par\vspace{2pt}%
    \textbullet\ Register kasir per outlet untuk integrasi PoS}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{CyanPrimary} 3. Transfer Stok}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Buat request transfer antar outlet/gudang\par\vspace{2pt}%
    \textbullet\ Item transfer dengan qty dan produk spesifik\par\vspace{2pt}%
    \textbullet\ Status tracking transfer (pending/approved/done)\par\vspace{2pt}%
    \textbullet\ Saldo stok otomatis disesuaikan setelah transfer}%

\end{minipage}%
\hfill%
\begin{minipage}[t]{0.485\textwidth}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{CyanPrimary} 4. Stock Opname}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Buat sesi opname per outlet/gudang\par\vspace{2pt}%
    \textbullet\ Input qty fisik per item dengan catatan\par\vspace{2pt}%
    \textbullet\ Kalkulasi selisih (stok sistem vs fisik) otomatis\par\vspace{2pt}%
    \textbullet\ Finalisasi opname mengunci perubahan\par\vspace{2pt}%
    \textbullet\ Penyesuaian stok (adjustment) berdasarkan hasil opname}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{CyanPrimary} 5. Purchase Order (PO Pembelian)}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Buat PO ke supplier dengan item detail\par\vspace{2pt}%
    \textbullet\ Status PO: draft, ordered, received, cancelled\par\vspace{2pt}%
    \textbullet\ Stok otomatis bertambah saat PO diterima\par\vspace{2pt}%
    \textbullet\ Manajemen supplier terintegrasi}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{CyanPrimary} 6. Penjualan \& Pembayaran (PoS)}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Sales order terintegrasi dengan stok (PoS module)\par\vspace{2pt}%
    \textbullet\ Multi-payment per transaksi penjualan\par\vspace{2pt}%
    \textbullet\ Shift kasir dengan kontrol kas masuk/keluar\par\vspace{2pt}%
    \textbullet\ Kelompok pelanggan \& harga segmentasi}%
