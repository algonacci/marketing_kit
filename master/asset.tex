\sectiontitle{FITUR UTAMA OMNIFLOW ASSET MANAGEMENT}

\noindent
\begin{minipage}[t]{0.485\textwidth}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{TealPrimary} 1. Registrasi \& Identifikasi Aset}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Asset tag unik (kode identifikasi internal)\par\vspace{2pt}%
    \textbullet\ Serial number, QR code per aset\par\vspace{2pt}%
    \textbullet\ Kategori aset (elektronik, kendaraan, furnitur, dll)\par\vspace{2pt}%
    \textbullet\ Status: available, assigned, borrowed, maintenance, retired\par\vspace{2pt}%
    \textbullet\ Kondisi: excellent, good, fair, damaged\par\vspace{2pt}%
    \textbullet\ Lokasi fisik saat ini tercatat}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{TealPrimary} 2. Penugasan Aset}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Assign ke karyawan, tim, atau lokasi tertentu\par\vspace{2pt}%
    \textbullet\ Handover notes saat serah terima\par\vspace{2pt}%
    \textbullet\ Tanggal mulai dan berakhir penugasan\par\vspace{2pt}%
    \textbullet\ Riwayat penugasan lengkap per aset\par\vspace{2pt}%
    \textbullet\ Custody type: employee / team / location}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{TealPrimary} 3. Peminjaman Aset}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Proses peminjaman dengan waktu mulai \& batas kembali\par\vspace{2pt}%
    \textbullet\ Checkout notes dan return notes\par\vspace{2pt}%
    \textbullet\ Status: ongoing, returned, overdue\par\vspace{2pt}%
    \textbullet\ Notifikasi overdue (aset belum dikembalikan)}%

\end{minipage}%
\hfill%
\begin{minipage}[t]{0.485\textwidth}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{TealPrimary} 4. Request \& Approval}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Tipe request: assignment, borrowing, return, status\_update, archive\par\vspace{2pt}%
    \textbullet\ Assign approver per request\par\vspace{2pt}%
    \textbullet\ Status: pending, approved, rejected, cancelled\par\vspace{2pt}%
    \textbullet\ Notes permintaan \& catatan reviewer\par\vspace{2pt}%
    \textbullet\ Timestamp reviewed\_at tercatat}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{TealPrimary} 5. Event Log Aset}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Setiap kejadian pada aset tercatat (event\_type)\par\vspace{2pt}%
    \textbullet\ Judul event, catatan, dan metadata JSON fleksibel\par\vspace{2pt}%
    \textbullet\ Telusuri riwayat lengkap setiap aset\par\vspace{2pt}%
    \textbullet\ Basis untuk laporan audit kapanpun}%
    \vspace{8pt}

    {\fontsize{10.5}{13}\selectfont\bfseries\color{TealPrimary} 6. Scan Token \& QR}
    \vspace{3pt}

    {\fontsize{9}{13}\selectfont\color{TextDark}%
    \textbullet\ Setiap aset punya scan token unik\par\vspace{2pt}%
    \textbullet\ QR code dapat dicetak dan ditempelkan ke aset\par\vspace{2pt}%
    \textbullet\ Scan QR langsung buka detail aset di browser\par\vspace{2pt}%
    \textbullet\ Update status aset cepat via scan}%
